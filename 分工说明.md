文件读写
// 文件操作基本函数
FILE *fp = fopen("students.dat", "wb+");
if (fp == NULL) {
    printf("文件打开失败!\n");
    return;
}

// 写入数据
fwrite(&student, sizeof(struct Student), 1, fp);

// 读取数据
fread(&student, sizeof(struct Student), 1, fp);

// 关闭文件
fclose(fp);
文件模式
"r": 只读
"w": 只写（覆盖）
"a": 追加
"r+": 读写
"w+": 读写（覆盖）
"a+": 读写（追加）
